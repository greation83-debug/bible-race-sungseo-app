// ==================================================================== // Supabase Edge Function for Notion API (페이지네이션 수정 버전) // 이 코드를 Supabase Edge Function (예: notion-proxy)에 배포하세요. // ==================================================================== import { serve } from "https://deno.land/std@0.168.0/http/server.ts"; // CORS 설정을 위한 헤더 (브라우저에서 직접 호출 허용) const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type", }; serve(async (req) => { // 1. CORS Preflight 요청 처리 (옵션 요청) if (req.method === "OPTIONS") { return new Response("ok", { headers: corsHeaders }); } try { // 2. 환경 변수에서 Notion 키 가져오기 (Supabase Secrets에 저장해야 함) const NOTION_KEY = Deno.env.get("NOTION_KEY"); const NOTION_DB_ID = Deno.env.get("NOTION_DB_ID"); if (!NOTION_KEY || !NOTION_DB_ID) { throw new Error("서버 환경 변수(Notion Key)가 설정되지 않았습니다."); } // 3. 클라이언트(HTML)에서 보낸 데이터 받기 const { tag, date } = await req.json(); if (!tag || !date) { return new Response(JSON.stringify({ error: "태그와 날짜 정보가 필요합니다." }), { headers: { ...corsHeaders, "Content-Type": "application/json" }, status: 400, }); } // 4. Notion API 호출: 데이터베이스 쿼리 console.log(`Querying Notion: Tag=${tag}, Date=${date}`); const queryResp = await fetch(`https://api.notion.com/v1/databases/${NOTION_DB_ID}/query`, { method: "POST", headers: { "Authorization": `Bearer ${NOTION_KEY}`, "Notion-Version": "2022-06-28", "Content-Type": "application/json", }, body: JSON.stringify({ filter: { property: "태그", multi_select: { contains: tag } }, sorts: [{ property: "날짜", direction: "ascending" }] }), }); if (!queryResp.ok) { throw new Error(`Notion DB Query Error: ${queryResp.status}`); } const queryData = await queryResp.json(); // 5. 날짜 매칭 로직 (서버에서 처리하여 로직 숨김) const matchedPage = queryData.results.find((page: any) => { let dateVal = null; // '게시날짜' 수식 속성 또는 '날짜' 속성 확인 if (page.properties['게시날짜']?.formula?.type === 'date') { dateVal = page.properties['게시날짜'].formula.date?.start; } if (!dateVal && page.properties['날짜']?.date) { dateVal = page.properties['날짜'].date.start; } if (!dateVal) return false; // "YYYY-MM-DD" 에서 "MM-DD"만 비교 return dateVal.slice(5) === date; }); if (!matchedPage) { return new Response(JSON.stringify({ title: null, text: `[알림] 날짜(${date})에 해당하는 말씀이 없습니다.` }), { headers: { ...corsHeaders, "Content-Type": "application/json" }, status: 200, }); } // 6. 페이지 블록(본문) 가져오기 - ★ 페이지네이션 추가 ★ const pageTitle = matchedPage.properties['이름']?.title[0]?.plain_text || "무제"; // ★ 오디오 URL 가져오기 (노션 DB에 '오디오URL' URL 속성이 있는 경우) let audioUrl = matchedPage.properties['오디오URL']?.url || null; // ★ Google Drive URL 자동 변환 if (audioUrl && audioUrl.includes('drive.google.com/file/d/')) { const match = audioUrl.match(/\/d\/([a-zA-Z0-9_-]+)/); if (match && match[1]) { audioUrl = `https://drive.google.com/uc?export=download&id=${match[1]}`; } } // 모든 블록을 담을 배열 let allBlocks: any[] = []; let hasMore = true; let nextCursor: string | null = null; // has_more가 false가 될 때까지 반복 요청 while (hasMore) { const url = nextCursor ? `https://api.notion.com/v1/blocks/${matchedPage.id}/children?start_cursor=${nextCursor}` : `https://api.notion.com/v1/blocks/${matchedPage.id}/children`; const blocksResp = await fetch(url, { method: "GET", headers: { "Authorization": `Bearer ${NOTION_KEY}`, "Notion-Version": "2022-06-28", }, }); if (!blocksResp.ok) { throw new Error("본문을 불러올 수 없습니다."); } const blocksData = await blocksResp.json(); // 결과 블록들을 배열에 추가 allBlocks = allBlocks.concat(blocksData.results); // 다음 페이지 여부 확인 hasMore = blocksData.has_more; nextCursor = blocksData.next_cursor; console.log(`Fetched ${blocksData.results.length} blocks, total: ${allBlocks.length}, has_more: ${hasMore}`); } // 7. 블록을 Markdown 텍스트로 변환 let fullText = ""; allBlocks.forEach((block: any) => { const type = block.type; if (block[type].rich_text) { const textContent = block[type].rich_text.map((t: any) => t.plain_text).join(''); if (type === 'heading_1') fullText += `# ${textContent}\n\n`; else if (type === 'heading_2') fullText += `## ${textContent}\n\n`; else if (type === 'heading_3') fullText += `### ${textContent}\n\n`; else if (type === 'bulleted_list_item') fullText += `• ${textContent}\n`; else if (type === 'numbered_list_item') fullText += `1. ${textContent}\n`; else if (type === 'quote') fullText += `> ${textContent}\n\n`; else fullText += `${textContent}\n\n`; } }); if (!fullText.trim()) { const aiSummary = matchedPage.properties['AI 요약']?.rich_text?.[0]?.plain_text; if (aiSummary) fullText = `[AI 요약]\n${aiSummary}`; else fullText = "본문 내용이 없습니다."; } // 8. 결과 반환 (오디오URL 포함) return new Response(JSON.stringify({ title: pageTitle, text: fullText, audioUrl }), { headers: { ...corsHeaders, "Content-Type": "application/json" }, status: 200, }); } catch (error) { return new Response(JSON.stringify({ error: error.message }), { headers: { ...corsHeaders, "Content-Type": "application/json" }, status: 500, }); } });
